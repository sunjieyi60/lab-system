## 流程概述

班牌系统包含三个核心业务流程：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                          班牌业务流程                                    │
├─────────────────────────────────────────────────────────────────────────┤
│                                                                         │
│  ┌─────────────┐     ┌─────────────┐     ┌─────────────────────────┐   │
│  │  流程1:握手  │────▶│  流程2:配置  │────▶│  流程3:人脸录入         │   │
│  └─────────────┘     └─────────────┘     └─────────────────────────┘   │
│                                                                         │
│  1. 班牌上电发送 WS Handshake                                           │
│  2. 服务端下发默认配置 (COMMAND_CONFIG_PUSH_RESP)                        │
│  3. 客户端应用配置，服务端 bindClassTimeTable() 完成通道绑定              │
│  4. 管理员通过 HTTP 接口配置班牌                                         │
│  5. 服务端下发配置更新，等待客户端响应                                   │
│  6. 管理员下发人脸照片 Bitmap 包                                         │
│  7. 班牌本地验证并录入人脸库，返回结果                                   │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘
```

### 1. ImPacket - 通信消息包

```java
package xyz.jasenon.lab.core;

import org.tio.core.intf.Packet;
import xyz.jasenon.lab.core.packets.Command;

/**
 * J-IM 通信消息包基类
 * 基于 t-io 框架的 Packet 扩展，支持加密、压缩、同步序列号等特性
 * 
 * @author WChao
 */
public class ImPacket extends Packet implements ImConst {
    
    private static final long serialVersionUID = 2000118564569232098L;
    
    /**
     * 包状态码
     */
    protected Status status;
    
    /**
     * 消息体（字节数组）
     */
    protected byte[] body;
    
    /**
     * 消息命令类型
     */
    private Command command;
    
    /**
     * IM 通道上下文（非序列化字段）
     */
    protected ImChannelContext imChannelContext;

    public ImPacket() {}
    
    public ImPacket(byte[] body) {
        this.body = body;
    }
    
    public ImPacket(Command command, byte[] body) {
        this(body);
        this.setCommand(command);
    }
    
    public ImPacket(Command command) {
        this(command, null);
    }

    // ========== 协议头编码/解码方法 ==========
    
    /**
     * 编码加密标识位
     */
    public static byte encodeEncrypt(byte bs, boolean isEncrypt) {
        if (isEncrypt) {
            return (byte) (bs | Protocol.FIRST_BYTE_MASK_ENCRYPT);
        } else {
            return (byte) (Protocol.FIRST_BYTE_MASK_ENCRYPT & 0b01111111);
        }
    }
    
    /**
     * 解码压缩标识
     */
    public static boolean decodeCompress(byte version) {
        return (Protocol.FIRST_BYTE_MASK_COMPRESS & version) != 0;
    }

    /**
     * 编码压缩标识位
     */
    public static byte encodeCompress(byte bs, boolean isCompress) {
        if (isCompress) {
            return (byte) (bs | Protocol.FIRST_BYTE_MASK_COMPRESS);
        } else {
            return (byte) (bs & (Protocol.FIRST_BYTE_MASK_COMPRESS ^ 0b01111111));
        }
    }

    /**
     * 解码同步序列号标识
     */
    public static boolean decodeHasSynSeq(byte maskByte) {
        return (Protocol.FIRST_BYTE_MASK_HAS_SYNSEQ & maskByte) != 0;
    }

    /**
     * 编码同步序列号标识位
     */
    public static byte encodeHasSynSeq(byte bs, boolean hasSynSeq) {
        if (hasSynSeq) {
            return (byte) (bs | Protocol.FIRST_BYTE_MASK_HAS_SYNSEQ);
        } else {
            return (byte) (bs & (Protocol.FIRST_BYTE_MASK_HAS_SYNSEQ ^ 0b01111111));
        }
    }

    /**
     * 计算消息头占用字节数
     */
    public int calcHeaderLength(boolean is4byteLength) {
        int ret = Protocol.LEAST_HEADER_LENGTH;
        if (is4byteLength) {
            ret += 2;
        }
        if (this.getSynSeq() > 0) {
            ret += 4;
        }
        return ret;
    }

    // ========== Getter/Setter ==========
    
    public Command getCommand() {
        return command;
    }

    public void setCommand(Command type) {
        this.command = type;
    }

    public byte[] getBody() {
        return body;
    }

    public void setBody(byte[] body) {
        this.body = body;
    }

    public Status getStatus() {
        return status;
    }

    public void setStatus(Status status) {
        this.status = status;
    }

    public ImChannelContext getImChannelContext() {
        return imChannelContext;
    }

    public void setImChannelContext(ImChannelContext imChannelContext) {
        this.imChannelContext = imChannelContext;
    }
    
    @Override
    public String logstr() {
        return this.command == null ? Command.COMMAND_UNKNOW.name() : this.command.name();
    }
}
```

package xyz.jasenon.lab.core.packets;

    COMMAND_GET_CLASS_TIME_TABLE_RESP(6),

    COMMAND_JOIN_GROUP_REQ(7),
    
    /** 加入群组响应 */
    COMMAND_JOIN_GROUP_RESP(8),
    
    /** 心跳请求 */
    COMMAND_HEARTBEAT_REQ(13),
    
    /** 心跳响应 */
    COMMAND_HEARTBEAT_RESP(14),
    
    /** 关闭请求 */
    COMMAND_CLOSE_REQ(15),
    
    /** 班牌设备配置信息下发 */
    COMMAND_CONFIG_PUSH_RESP(16);

    private final int value;

    private Command(int value) {
        this.value = value;
    }

    public final int getNumber() {
        return value;
    }

    public static Command valueOf(int value) {
        return forNumber(value);
    }

    public static Command forNumber(int value) {
        for (Command command : Command.values()) {
            if (command.getNumber() == value) {
                return command;
            }
        }
        return null;
    }
}

package xyz.jasenon.lab.core.packets;

import com.alibaba.fastjson.JSONObject;
import xyz.jasenon.lab.core.utils.JsonKit;

import java.io.Serializable;

/**
 * 消息基类
 * 所有请求/响应消息体的父类
 * 
 * @author WChao
 */
public class Message implements Serializable {

    private static final long serialVersionUID = -6375331164604259933L;

    /**
     * 消息创建时间（Unix 时间戳）
     */
    protected Long createTime;

    /**
     * 消息 ID，全局唯一
     */
    protected String id;

    /**
     * 消息命令码
     */
    protected Integer cmd;

    /**
     * 扩展参数字段（JSON 格式）
     */
    protected JSONObject extras;

    // ========== Getter/Setter ==========

    public Long getCreateTime() {
        return createTime;
    }

    public void setCreateTime(Long createTime) {
        this.createTime = createTime;
    }

    public String getId() {
        return id;
    }

    public void setId(String requestId) {
        this.id = requestId;
    }

    public Integer getCmd() {
        return cmd;
    }

    public void setCmd(Integer cmd) {
        this.cmd = cmd;
    }

    public JSONObject getExtras() {
        return extras;
    }

    public void setExtras(JSONObject extras) {
        this.extras = extras;
    }

    // ========== 序列化方法 ==========

    public String toJsonString() {
        return JsonKit.toJSONString(this);
    }

    public byte[] toByte() {
        return JsonKit.toJsonBytes(this);
    }

    // ========== Builder 模式 ==========

    public abstract static class Builder<T extends Message, B extends Builder<T, B>> {
        protected Long createTime;
        protected String id;
        protected Integer cmd;
        protected JSONObject extras;

        private B theBuilder = this.getThis();

        protected abstract B getThis();

        public B setCreateTime(Long createTime) {
            this.createTime = createTime;
            return theBuilder;
        }

        public B setId(String id) {
            this.id = id;
            return theBuilder;
        }

        public B setCmd(Integer cmd) {
            this.cmd = cmd;
            return theBuilder;
        }

        public B addExtra(String key, Object value) {
            if (null == value) {
                return theBuilder;
            } else {
                if (null == extras) {
                    this.extras = new JSONObject();
                }
                this.extras.put(key, value);
                return theBuilder;
            }
        }

        public abstract T build();
    }
}
